# User Block Library Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a personal/team library of saved blocks to Lemnity Box Editor — users save GrapesJS and Zero Block sections, preview them with CSS-scaled HTML, and insert them back onto the canvas.

**Architecture:** New `UserSavedBlock` Prisma model → service layer `lib/user-saved-blocks.ts` → 5 API routes → two UI components (`LemnityBoxUserBlocksPanel` + `LemnityBoxSaveBlockDialog`) wired into `lemnity-box-canvas-editor.tsx` (new `<aside>` panel + GrapesJS toolbar button) and `zb-top-bar.tsx` (new "В библиотеку" button).

**Tech Stack:** Prisma ORM, Vitest (TDD), Next.js App Router API routes, React + Tailwind (panel/dialog), GrapesJS `Model.prototype.initToolbar` monkey-patch

  ```typescript
  import { prisma } from "@/lib/prisma";

  export type UserSavedBlockMeta = {
    id: string;
    name: string;
    blockType: "grapesjs" | "zero";
    scope: "personal" | "team";
    createdAt: string;
  };

  export type UserSavedBlockFull = UserSavedBlockMeta & {
    htmlContent: string;
    cssContent: string;
  };

  export type CreateUserBlockInput = {
    userId: string;
    name: string;
    blockType: "grapesjs" | "zero";
    htmlContent: string;
    cssContent: string;
    teamProjectId?: string | null;
  };

  const PERSONAL_LIMIT = 200;
  const TEAM_LIMIT = 500;

  function toMeta(row: {
    id: string;
    name: string;
    blockType: string;
    teamProjectId: string | null;
    createdAt: Date;
  }): UserSavedBlockMeta {
    return {
      id: row.id,
      name: row.name,
      blockType: row.blockType as "grapesjs" | "zero",
      scope: row.teamProjectId ? "team" : "personal",
      createdAt: row.createdAt.toISOString(),
    };
  }

  export async function listUserBlocks(
    userId: string,
    projectId?: string,
  ): Promise<UserSavedBlockMeta[]> {
    const where = projectId
      ? { OR: [{ userId, teamProjectId: null }, { teamProjectId: projectId }] }
      : { userId, teamProjectId: null };

    const rows = await prisma.userSavedBlock.findMany({
      where,
      orderBy: { createdAt: "desc" },
      select: { id: true, name: true, blockType: true, teamProjectId: true, createdAt: true },
    });

    return rows.map(toMeta);
  }

  export async function createUserBlock(
    input: CreateUserBlockInput,
  ): Promise<UserSavedBlockMeta> {
    const { userId, name, blockType, htmlContent, cssContent, teamProjectId } = input;

    const countWhere = teamProjectId
      ? { teamProjectId }
      : { userId, teamProjectId: null };
    const limit = teamProjectId ? TEAM_LIMIT : PERSONAL_LIMIT;

    const count = await prisma.userSavedBlock.count({ where: countWhere });
    if (count >= limit) {
      throw new Error(`Block library limit reached (${limit})`);
    }

    const created = await prisma.userSavedBlock.create({
      data: { userId, name, blockType, htmlContent, cssContent, teamProjectId: teamProjectId ?? null },
      select: { id: true, name: true, blockType: true, teamProjectId: true, createdAt: true },
    });

    return toMeta(created);
  }

  export async function getUserBlockById(
    id: string,
    userId: string,
  ): Promise<UserSavedBlockFull | null> {
    const row = await prisma.userSavedBlock.findFirst({
      where: { id, userId },
    });
    if (!row) return null;
    return {
      ...toMeta(row),
      htmlContent: row.htmlContent,
      cssContent: row.cssContent,
    };
  }

  export async function renameUserBlock(
    id: string,
    userId: string,
    name: string,
  ): Promise<boolean> {
    const row = await prisma.userSavedBlock.findFirst({
      where: { id, userId },
      select: { id: true },
    });
    if (!row) return false;
    await prisma.userSavedBlock.update({ where: { id }, data: { name } });
    return true;
  }

  export async function deleteUserBlock(id: string, userId: string): Promise<boolean> {
    const row = await prisma.userSavedBlock.findFirst({
      where: { id, userId },
      select: { id: true },
    });
    if (!row) return false;
    await prisma.userSavedBlock.delete({ where: { id } });
    return true;
  }
  ```

  ```typescript
  import { type NextRequest } from "next/server";
  import { apiOk, apiError, apiGuardError } from "@/lib/api-response";
  import { requireDbUser } from "@/lib/auth-guards";
  import { withApiLogging } from "@/lib/with-api-logging";
  import { parseBody } from "@/lib/api-schemas";
  import { z } from "zod";
  import { listUserBlocks, createUserBlock } from "@/lib/user-saved-blocks";

  export const dynamic = "force-dynamic";
  export const runtime = "nodejs";

  const createSchema = z.object({
    name: z.string().min(1).max(100),
    blockType: z.enum(["grapesjs", "zero"]),
    htmlContent: z.string().min(1),
    cssContent: z.string().optional().default(""),
    teamProjectId: z.string().optional(),
  });

  async function GET(req: NextRequest) {
    const guard = await requireDbUser(req);
    if (guard.status !== 200) return apiGuardError(guard);
    const { user } = guard;

    const projectId = new URL(req.url).searchParams.get("projectId") ?? undefined;
    const blocks = await listUserBlocks(user.id, projectId);
    return apiOk({ blocks });
  }

  async function POST(req: NextRequest) {
    const guard = await requireDbUser(req);
    if (guard.status !== 200) return apiGuardError(guard);
    const { user } = guard;

    const parsed = await parseBody(req, createSchema);
    if (!parsed.success) return apiError(parsed.error, 400);

    try {
      const block = await createUserBlock({ userId: user.id, ...parsed.data });
      return apiOk({ block }, 201);
    } catch (err) {
      const msg = err instanceof Error ? err.message : "Unexpected error";
      if (msg.includes("limit")) return apiError(msg, 429);
      return apiError(msg, 500);
    }
  }

  export const GET = withApiLogging("/api/user-blocks", GET);
  export const POST = withApiLogging("/api/user-blocks", POST);
  ```

  > **Note:** `withApiLogging` wraps the handler: signature is `withApiLogging(pathname, handler)`. The exports must shadow the function names — see how `app/api/projects/[id]/snapshots/route.ts` does it.

  ```typescript
  import { type NextRequest } from "next/server";
  import { apiOk, apiError, apiGuardError } from "@/lib/api-response";
  import { requireDbUser } from "@/lib/auth-guards";
  import { withApiLogging } from "@/lib/with-api-logging";
  import { parseBody } from "@/lib/api-schemas";
  import { z } from "zod";
  import { getUserBlockById, renameUserBlock, deleteUserBlock } from "@/lib/user-saved-blocks";

  export const dynamic = "force-dynamic";
  export const runtime = "nodejs";

  const renameSchema = z.object({ name: z.string().min(1).max(100) });

  async function GET(req: NextRequest, { params }: { params: Promise<{ id: string }> }) {
    const guard = await requireDbUser(req);
    if (guard.status !== 200) return apiGuardError(guard);
    const { user } = guard;

    const { id } = await params;
    const block = await getUserBlockById(id, user.id);
    if (!block) return apiError("Not found", 404);
    return apiOk({ block });
  }

  async function PATCH(req: NextRequest, { params }: { params: Promise<{ id: string }> }) {
    const guard = await requireDbUser(req);
    if (guard.status !== 200) return apiGuardError(guard);
    const { user } = guard;

    const { id } = await params;
    const parsed = await parseBody(req, renameSchema);
    if (!parsed.success) return apiError(parsed.error, 400);

    const ok = await renameUserBlock(id, user.id, parsed.data.name);
    if (!ok) return apiError("Not found", 404);
    return apiOk({ ok: true });
  }

  async function DELETE(req: NextRequest, { params }: { params: Promise<{ id: string }> }) {
    const guard = await requireDbUser(req);
    if (guard.status !== 200) return apiGuardError(guard);
    const { user } = guard;

    const { id } = await params;
    const ok = await deleteUserBlock(id, user.id);
    if (!ok) return apiError("Not found", 404);
    return new Response(null, { status: 204 });
  }

  export const GET = withApiLogging("/api/user-blocks/[id]", GET);
  export const PATCH = withApiLogging("/api/user-blocks/[id]", PATCH);
  export const DELETE = withApiLogging("/api/user-blocks/[id]", DELETE);
  ```

  > **Note on DELETE 204:** The spec requires 204 No Content with no body. Use `new Response(null, { status: 204 })` directly for this case — `apiOk()` sends 200 with JSON.

          blockType,
          htmlContent,
          cssContent,

            autoFocus
            type="text"
            value={name}
            onChange={(e) => setName(e.target.value)}
            onKeyDown={(e) => { if (e.key === "Enter") handleSave(); if (e.key === "Escape") onClose(); }}

  import type { Editor } from "grapesjs";

  const SAVE_BLOCK_CMD = "lemnity-save-block";
  const SAVE_BLOCK_ID = "lemnity-save-block-btn";

  const SAVE_SVG =
    '<svg viewBox="0 0 24 24" aria-hidden="true" width="16" height="16"><path fill="currentColor" d="M20 6H4a1 1 0 0 0-1 1v10a1 1 0 0 0 1 1h16a1 1 0 0 0 1-1V7a1 1 0 0 0-1-1zm-9 9H7v-4h4v4zm6 0h-4v-4h4v4zM4 4h16a1 1 0 0 1 0 2H4a1 1 0 0 1 0-2z"/></svg>';

  type ToolbarRow = {
    id?: string;
    label?: unknown;
    command?: unknown;
    attributes?: Record<string, string>;
  };

  /**
   * Registers the "🗂 Сохранить" toolbar button on GrapesJS sections.
   * Monkey-patches Model.prototype.initToolbar (same pattern as lemnity-box-toolbar-sibling-moves.ts).
   * The command fires the onSaveBlock callback with the selected component's HTML and CSS.
   */
  export function registerLemnityBoxToolbarSaveBlock(
    editor: Editor,
    onSaveBlock: (htmlContent: string, cssContent: string) => void,
  ): void {
    editor.Commands.add(SAVE_BLOCK_CMD, {
      run(ed: Editor) {
        const selected = ed.getSelected();
        if (!selected) return;
        const html = selected.toHTML?.() ?? "";
        const allCss = ed.getCss({ keepUnusedStyles: false }) ?? "";
        // Pass only the CSS that pertains to this component (full editor CSS as fallback)
        onSaveBlock(html, allCss);
      },
    });

    const dc = editor.DomComponents;
    const type = dc.getType("default");
    const Model = type?.model as { prototype: { initToolbar: () => void } } | undefined;
    if (!Model?.prototype?.initToolbar) return;

    const proto = Model.prototype;
    const original = proto.initToolbar;

    proto.initToolbar = function patchedSaveBlockToolbar(this: {
      get: (k: string) => unknown;
      set: (k: string, v: unknown) => void;
    }) {
      original.call(this);
      const raw = this.get("toolbar");
      if (!Array.isArray(raw)) return;

      const tb = raw.slice() as ToolbarRow[];
      if (tb.some((b) => b.id === SAVE_BLOCK_ID)) {
        this.set("toolbar", tb);
        return;
      }

      // Insert before the delete button (last in the array)
      tb.splice(tb.length - 1, 0, {
        id: SAVE_BLOCK_ID,
        label: SAVE_SVG,
        command: SAVE_BLOCK_CMD,
        attributes: { title: "Сохранить блок в библиотеку" },
      });

      this.set("toolbar", tb);
    };
  }

### Known Constraints

- **No `requireProjectAccess`** — `lib/auth-guards.ts` does not export this. Use `requireDbUser()` only. Team block access is implicitly controlled by `userId` ownership and `teamProjectId`. The service layer scopes queries by `userId` for personal blocks. For team blocks, any authenticated user can read blocks with a given `teamProjectId` — this is acceptable per the spec (all project team members).
- **`withApiLogging` signature** — pathname is the FIRST argument: `withApiLogging("/api/user-blocks", handler)`.
- **`import { prisma } from "@/lib/prisma"`** — named export, not default.
- **`params` in Next.js 15** — must be awaited: `const { id } = await params`.
- **Zero block ID generation** — uses `Math.random().toString(36).slice(2, 10)`, not `cuid()`. Matches existing pattern in canvas editor.
